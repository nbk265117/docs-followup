CREATE TYPE type_point_vente AS ENUM (
    'AGENCE_CITADINE', 'AGENCE_ETENDARD', 'AGENCE_CONTACT', 'AGENCE_EXCLUSIVE',

    'OBSERVATEUR', 'CIBLE', 'DOUBLE_ROLE', 'ADMINISTRATEUR'

-- 2. TABLES ORGANISATION

CREATE TABLE region (
    id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code        VARCHAR(10) UNIQUE NOT NULL,
    nom         VARCHAR(100) NOT NULL,
    niveau      CHAR(2) DEFAULT 'N5',
    actif       BOOLEAN DEFAULT TRUE,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE sous_region (
    id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code        VARCHAR(10) UNIQUE NOT NULL,
    nom         VARCHAR(100) NOT NULL,
    niveau      CHAR(2) DEFAULT 'N4',
    region_id   UUID NOT NULL REFERENCES region(id),
    actif       BOOLEAN DEFAULT TRUE,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE groupe (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code            VARCHAR(10) UNIQUE NOT NULL,
    nom             VARCHAR(100) NOT NULL,
    niveau          CHAR(2) DEFAULT 'N3',
    sous_region_id  UUID NOT NULL REFERENCES sous_region(id),
    actif           BOOLEAN DEFAULT TRUE,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE point_de_vente (
    id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code        VARCHAR(10) UNIQUE NOT NULL,
    nom         VARCHAR(150) NOT NULL,
    type        type_point_vente NOT NULL,
    groupe_id   UUID NOT NULL REFERENCES groupe(id),
    niveau      CHAR(2) DEFAULT 'N2',
    actif       BOOLEAN DEFAULT TRUE,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE desk (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code            VARCHAR(10) UNIQUE NOT NULL,
    nom             VARCHAR(100) NOT NULL,
    niveau          CHAR(2) DEFAULT 'N1',
    point_vente_id  UUID REFERENCES point_de_vente(id),
    actif           BOOLEAN DEFAULT TRUE,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

    code        VARCHAR(20) PRIMARY KEY,

    actif       BOOLEAN DEFAULT TRUE

CREATE TABLE role_followup (
    id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code        VARCHAR(30) UNIQUE NOT NULL,
    libelle     VARCHAR(100) NOT NULL,

    description TEXT

CREATE TABLE mapping_fonction_role (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    UNIQUE(fonction_asal_code, role_followup_id)

CREATE TABLE utilisateur_followup (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    matricule       VARCHAR(20) UNIQUE NOT NULL,
    nom             VARCHAR(100) NOT NULL,
    prenom          VARCHAR(100),
    email           VARCHAR(150),
    telephone       VARCHAR(20),
    fonction_code   VARCHAR(20) REFERENCES fonction_asal(code),
    role_id         UUID REFERENCES role_followup(id),
    point_vente_id  UUID REFERENCES point_de_vente(id),
    keycloak_id     VARCHAR(50),
    actif           BOOLEAN DEFAULT TRUE,
    date_sync_asal  TIMESTAMP,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    dernier_acces   TIMESTAMP

CREATE TABLE affectation_animateur (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    type_rattachement   type_rattachement NOT NULL,
    region_id           UUID REFERENCES region(id),
    groupe_id           UUID REFERENCES groupe(id),
    point_vente_id      UUID REFERENCES point_de_vente(id),
    date_debut          DATE NOT NULL DEFAULT CURRENT_DATE,
    date_fin            DATE,
    CONSTRAINT chk_affectation CHECK (
        (region_id IS NOT NULL AND groupe_id IS NULL AND point_vente_id IS NULL) OR
        (region_id IS NULL AND groupe_id IS NOT NULL AND point_vente_id IS NULL) OR
        (region_id IS NULL AND groupe_id IS NULL AND point_vente_id IS NOT NULL)
    )
);

CREATE TABLE modele_observation (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    code            VARCHAR(20) UNIQUE NOT NULL,  -- Ex: OBS_CAE, OBS_GE, OBS_CAC
    libelle         VARCHAR(100) NOT NULL,
    marche          type_marche NOT NULL,
    cible_role_id   UUID REFERENCES role_followup(id),
    actif           BOOLEAN DEFAULT TRUE,
    version         INTEGER DEFAULT 1,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE question (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    modele_id       UUID NOT NULL REFERENCES modele_observation(id) ON DELETE CASCADE,
    numero          INTEGER NOT NULL,
    libelle         TEXT NOT NULL,
    type_reponse    type_reponse NOT NULL DEFAULT 'BOOLEAN',
    obligatoire     BOOLEAN DEFAULT TRUE,
    ordre           INTEGER NOT NULL,
    UNIQUE(modele_id, numero)
);

CREATE TABLE observation (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    modele_id           UUID NOT NULL REFERENCES modele_observation(id),
    observateur_id      UUID NOT NULL REFERENCES utilisateur_followup(id),
    cible_id            UUID NOT NULL REFERENCES utilisateur_followup(id),
    point_vente_id      UUID NOT NULL REFERENCES point_de_vente(id),
    statut              statut_observation NOT NULL DEFAULT 'SOUMISE',
    date_soumission     TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    date_annulation     TIMESTAMP,
    motif_annulation    TEXT,
    commentaire_general TEXT,
    created_at          TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT chk_annulation CHECK (
        statut = 'ANNULEE' AND date_annulation IS NOT NULL OR
        statut = 'SOUMISE' AND date_annulation IS NULL
    )
);

CREATE TABLE reponse_observation (
    id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
